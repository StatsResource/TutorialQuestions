\begin{itemize}
	\item[(i)] (1 Mark) Calculate a point estimate $\hat{p}$ for the proportion of plants that grow faster due to the dosage. 									 
		
	\item[(ii)]  Find a 95\% confidence interval for the proportion. 					
\end{itemize}



\end{enumerate}
